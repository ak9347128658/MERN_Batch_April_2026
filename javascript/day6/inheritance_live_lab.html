Inheritance live lab with Teacher and Student extending Person

// PARENT class
class Person {
  constructor(name, age) { this.name = name; this.age = age; }
  greet() { return `Hello! I am ${this.name}`; }
  introduce() { return `I am ${this.name}, age ${this.age}`; }
}

// CHILD class — inherits Person + adds its own
class Teacher extends Person {
  constructor(name, age, subject) {
    super(name, age); // calls Person's constructor
    this.subject = subject;
  }
  teach() { return `${this.name} teaches ${this.subject}`; }
}

class Student extends Person {
  constructor(name, age, score) {
    super(name, age); // calls Person's constructor
    this.score = score;
  }
  study() { return `${this.name} is studying!`; }
  getGrade() { return this.score >= 90 ? "A" : "B"; }
}
