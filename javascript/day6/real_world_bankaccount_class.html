Real world BankAccount class demonstration with deposit withdraw and transactions

class BankAccount {
  constructor(owner, initialBalance = 0) {
    this.owner = owner;
    this.balance = initialBalance;
    this.transactions = [];
  }
  deposit(amount) {
    if (amount <= 0) throw "Amount must be positive";
    this.balance += amount;
    this.transactions.push({ type:"credit", amount });
  }
  withdraw(amount) {
    if (amount > this.balance) throw "Insufficient balance";
    this.balance -= amount;
    this.transactions.push({ type:"debit", amount });
  }
  get summary() {
    return `${this.owner} — Balance: ₹${this.balance}`;
  }
}
