// 3 ways to loop through an object
const user = { name:"Ali", age:22, city:"Hyderabad", score:88 };

// 1. for...in loop
for (let key in user) { console.log(key, "→", user[key]); }

// 2. Object.keys() + forEach
Object.keys(user).forEach(k => console.log(k, user[k]));

// 3. Object.entries() — best for both key and value
Object.entries(user).forEach(([k, v]) => console.log(k, v));
