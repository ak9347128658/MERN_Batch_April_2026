Array creation and index access interactive visualizer

// Creating an array — use square brackets [ ]
let students = ["Alice", "Bob", "Charlie", "Diana", "Eve"];

// Accessing items — use the index number
students[0] // "Alice" ← first
students[2] // "Charlie" ← third
students[students.length - 1] // "Eve" ← last (always!)
students.length // 5 (total count)
