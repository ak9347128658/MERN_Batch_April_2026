All nested array operations read write add row remove item loop

Read — access items
// Read one item
grid[1][2] // "Tom"

// Read whole row
grid[0] // ["Ali","Sara","Ravi"]

// Read total rows
grid.length // 3
—
Write — change a value
// Change one item
grid[0][0] = "Ahmed";

// Change whole row
grid[1] = ["X", "Y", "Z"];
—
Add — push a new row
// Add a whole new row
grid.push(["New1", "New2", "New3"]);

// Add to existing row
grid[0].push("ExtraItem");
—
Loop — forEach on nested
// forEach on each inner array
grid.forEach((row, r) => {
 row.forEach((item, c) => {
  console.log(r, c, item);
 });
});
grid.flat() collapses nested → one flat array
