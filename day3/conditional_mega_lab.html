// 1. if/else — most readable, use for complex logic
if (score >= 60) { console.log("Pass"); }
else { console.log("Fail"); }

// 2. switch — best for exact value matching
switch(true) {
  case score >= 60: console.log("Pass"); break;
  default: console.log("Fail");
}

// 3. ternary — shortest, use for simple one-liners
let result = score >= 60 ? "Pass" : "Fail";
65
All 3 give the same answer: "Pass"
